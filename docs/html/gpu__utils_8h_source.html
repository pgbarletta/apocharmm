61 static __device__ double atomicAdd(double *address, double val) {
62  unsigned long long int *address_as_ull = (unsigned long long int *)address;
63  unsigned long long int old = *address_as_ull, assumed;
64  do {
65  assumed = old;
66  old = atomicCAS(address_as_ull, assumed, __double_as_longlong(val + __longlong_as_double(assumed)));

106 static __device__ float atomicMin(float *address, float val) {
107  int *address_as_i = (int *)address;
108  int old = *address_as_i, assumed;
109  do {
110  assumed = old;
111  old = atomicCAS(address_as_i, assumed, __float_as_int(fminf(val, __int_as_float(assumed))));
112  } while (assumed != old);

142 // Faster ERFC approximation courtesy of Norbert Juffa. NVIDIA Corporation
143 static __forceinline__ __device__ float fasterfc(float a) {
144  /* approximate log(erfc(a)) with rel. error < 7e-9 */
145  float t, x = a;
146  t = (float)-1.6488499458192755E-006;
147  t = __internal_fmad(t, x, (float)2.9524665006554534E-005);
148  t = __internal_fmad(t, x, (float)-2.3341951153749626E-004);
149  t = __internal_fmad(t, x, (float)1.0424943374047289E-003);
150  t = __internal_fmad(t, x, (float)-2.5501426008983853E-003);
151  t = __internal_fmad(t, x, (float)3.1979939710877236E-004);
152  t = __internal_fmad(t, x, (float)2.7605379075746249E-002);
153  t = __internal_fmad(t, x, (float)-1.4827402067461906E-001);
154  t = __internal_fmad(t, x, (float)-9.1844764013203406E-001);
155  t = __internal_fmad(t, x, (float)-1.6279070384382459E+000);
156  t = t * x;
157  return exp2f(t);
158 }

292 // Calculates box shift
293 //
294 // ish = (imx+1) + 3*(imy+1) + 9*(imz+1) = 0...26
295 //
296 template <typename T>
297 __forceinline__ __device__ __host__ void calc_box_shift(int ish, const T boxx, const T boxy, const T boxz, T &shx,
298  T &shy, T &shz) {
299  shz = (ish / 9 - 1) * boxz;
300  ish -= (ish / 9) * 9;
301  shy = (ish / 3 - 1) * boxy;
302  ish -= (ish / 3) * 3;
303  shx = (ish - 1) * boxx;
304 }

308 __forceinline__ __device__ int calc_ishift(const float4 xyzq_i, const float4 xyzq_j, const float3 half_box) {
309  float3 dxyz;
310  dxyz.x = xyzq_i.x - xyzq_j.x;
311  dxyz.y = xyzq_i.y - xyzq_j.y;
312  dxyz.z = xyzq_i.z - xyzq_j.z;
313 
314  int3 is;
315  is.x = 0;
316  is.y = 0;
317  is.z = 0;
318 
319  // is = -1, 0, 1
320  if (dxyz.x >= half_box.x) {
321  is.x = -1;
322  } else if (dxyz.x < -half_box.x) {
323  is.x = 1;
324  }
325 
326  if (dxyz.y >= half_box.y) {
327  is.y = -1;
328  } else if (dxyz.y < -half_box.y) {
329  is.y = 1;
330  }
331 
332  if (dxyz.z >= half_box.z) {
333  is.z = -1;
334  } else if (dxyz.z < -half_box.z) {
335  is.z = 1;
336  }
337 
338  return is.x + 1 + (is.y + 1) * 3 + (is.z + 1) * 9;
339 }
