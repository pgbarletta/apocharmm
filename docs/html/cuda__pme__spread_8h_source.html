10 // body of spread_charge_ortho_ORDER(_block)
11 //
12 // Spreads the charge on the grid. Calculates theta and dtheta on the fly
13 // blockDim.x = Number of atoms each block loads
14 // blockDim.y*blockDim.x/64 = Number of atoms we spread at once
15 #define SQUARE_ORDER (ORDER * ORDER)
16 
17 #if ORDER == 6
18 #define CUBE_ORDER 224
19 #else
20 #define CUBE_ORDER (ORDER * ORDER * ORDER)
21 #endif
22 
23 (const float4 *xyzq, const int ncoord, const float recip11, const float recip22, const float recip33, const int nfftx,
24  const int nffty, const int nfftz,
25 #ifdef DOMDEC_MSLDPME
26  const float *bixlam, const int *blockIndexes,
27 #endif
28  AT *data) {
29 
30  __shared__ int sh_ix[32];
31  __shared__ int sh_iy[32];
32  __shared__ int sh_iz[32];
33  __shared__ float sh_q[32];
34 
35  __shared__ float sh_thetax[ORDER * 32];
36  __shared__ float sh_thetay[ORDER * 32];
37  __shared__ float sh_thetaz[ORDER * 32];
38 
39  // Process atoms pos to pos_end-1
40  const unsigned int pos = blockIdx.x * blockDim.x + threadIdx.x,
41  pos_end = min((blockIdx.x + 1) * blockDim.x, ncoord);
42 
43  if (pos < pos_end && threadIdx.y == 0) {
44  float4 xyzqi = xyzq[pos];
45 
46  float x = xyzqi.x;
47  float y = xyzqi.y;
48  float z = xyzqi.z;
49  float q = xyzqi.w;
50 
51 #ifdef DOMDEC_MSLDPME
52  int bii = blockIndexes[pos];
53  bii &= 0xffff; // First 16 bits are site index
54  if (bii > 0) {
55  q *= bixlam[bii];
56  }
57 #endif
58 
59  sh_q[threadIdx.x] = q;
60 
61  float w;
62 
63  w = x * recip11 + 2.0f;
64  float frx = (float)(nfftx * (w - (floorf(w + 0.5f) - 0.5f)));
65  w = y * recip22 + 2.0f;
66  float fry = (float)(nffty * (w - (floorf(w + 0.5f) - 0.5f)));
67  w = z * recip33 + 2.0f;
68  float frz = (float)(nfftz * (w - (floorf(w + 0.5f) - 0.5f)));
69 
70  int frxi = (int)frx;
71  int fryi = (int)fry;
72  int frzi = (int)frz;
73 
74  sh_ix[threadIdx.x] = frxi;
75  sh_iy[threadIdx.x] = fryi;
76  sh_iz[threadIdx.x] = frzi;
77 
78  float wx = frx - (float)frxi;
79  float wy = fry - (float)fryi;
80  float wz = frz - (float)frzi;
81 
82  float theta[ORDER];
83 
84  calc_one_theta<float, ORDER>(wx, theta);
85 #pragma unroll
86  for (int i = 0; i < ORDER; ++i) sh_thetax[threadIdx.x * ORDER + i] = theta[i];
87 
88  calc_one_theta<float, ORDER>(wy, theta);
89 #pragma unroll
90  for (int i = 0; i < ORDER; ++i) sh_thetay[threadIdx.x * ORDER + i] = theta[i];
91 
92  calc_one_theta<float, ORDER>(wz, theta);
93 #pragma unroll
94  for (int i = 0; i < ORDER; ++i) sh_thetaz[threadIdx.x * ORDER + i] = theta[i];
95  }
96 
97  __syncthreads();
98 
99  // Grid point location, values of (ix0, iy0, iz0) are in range 0..5
100  // NOTE: Only tid=0...215 do any computation for order 6
101  const int tid = (threadIdx.x + threadIdx.y * blockDim.x) % CUBE_ORDER;
102 #if ORDER == 6
103  if (tid >= 216) return;
104 #endif
105 
106  const int x0 = tid % ORDER;
107  const int y0 = (tid / ORDER) % ORDER;
108  const int z0 = tid / SQUARE_ORDER;
109 
110  // Loop over atoms pos..pos_end-1
111  int iadd = blockDim.x * blockDim.y / CUBE_ORDER;
112  int i = (threadIdx.x + threadIdx.y * blockDim.x) / CUBE_ORDER;
113  int iend = pos_end - blockIdx.x * blockDim.x;
114  for (; i < iend; i += iadd) {
115  int x = sh_ix[i] + x0;
116  int y = sh_iy[i] + y0;
117  int z = sh_iz[i] + z0;
118 
119  float q = sh_q[i];
120 
121  if (x >= nfftx) x -= nfftx;
122  if (y >= nffty) y -= nffty;
123  if (z >= nfftz) z -= nfftz;
124 
125  // Get position on the grid
126  int ind = x + nfftx * (y + nffty * z);
127 
128  // Here we unroll the CUBE_ORDER loop with CUBE_ORDER threads.
129  // Calculate interpolated charge value and store it to global memory
130  write_grid<AT>(q * sh_thetax[i * ORDER + x0] * sh_thetay[i * ORDER + y0] * sh_thetaz[i * ORDER + z0], ind,
131  data);
132  }
133 }
134 #undef SQUARE_ORDER
135 #undef CUBE_ORDER
