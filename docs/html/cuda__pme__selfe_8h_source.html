10 // calc_self_energy_kernel body
11 // Calculates self energy
12 // kappa_ccelec_sqrtpi = kappa*ccelec/sqrt(pi)
13 (const int ncoord, const float4* xyzq, const double kappa_ccelec_sqrtpi,
14 #ifdef DOMDEC_MSLDPME
15  const float* bixlam, double* biflam, const int* blockIndexes,
16 #endif
17  double* __restrict__ energy_self) {
18 
19  // Shared memory
20  // Required space: blockDim.x*sizeof(double)
21  extern __shared__ double sh_q2[];
22 
23  int i = threadIdx.x + blockIdx.x * blockDim.x;
24  float q = 0.0f;
25  if (i < ncoord) {
26  // printf("%d\t%f\t%f\t%f\t%f\n", i, xyzq[i].x, xyzq[i].y, xyzq[i].z, xyzq[i].w);
27  q = xyzq[i].w;
28 #ifdef DOMDEC_MSLDPME
29  int bii = blockIndexes[i];
30  bii &= 0xffff; // First 16 bits are site index
31  if (bii > 0) {
32  atomicAdd(&biflam[bii], -2.0 * bixlam[bii] * q * q * kappa_ccelec_sqrtpi);
33  q *= bixlam[bii];
34  }
35 #endif
36  }
37  sh_q2[threadIdx.x] = q * q;
38  __syncthreads();
39  for (int d = 1; d < blockDim.x; d *= 2) {
40  int t = threadIdx.x + d;
41  double q2_val = (t < blockDim.x) ? sh_q2[t] : 0.0;
42  __syncthreads();
43  sh_q2[threadIdx.x] += q2_val;
44  __syncthreads();
45  }
46  if (threadIdx.x == 0) {
47  // printf("Thread : %d >> Updating energy_self by %f\n",i, q);
48  atomicAdd(energy_self, -sh_q2[0] * kappa_ccelec_sqrtpi);
49  // printf("Thread : %d >> to %.6f\n", i, *energy_self);
50  }
51 }
