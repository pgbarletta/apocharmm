10 // gather_force_ORDER_ortho_kernel body
11 // Gathers forces from the grid
12 // blockDim.x = Number of atoms each block loads
13 // blockDim.x*blockDim.y = Total number of threads per block
14 #define HALF_ORDER (ORDER / 2)
15 #define SQUARE_HALF (HALF_ORDER * HALF_ORDER)
16 #define CUBE_HALF (HALF_ORDER * SQUARE_HALF)
17 (const float4 *xyzq, const int ncoord, const int nfftx, const int nffty, const int nfftz, const int xsize,

19 #ifdef USE_TEXTURE_OBJECTS

21 #endif

23 #ifdef DOMDEC_MSLDPME

26 #endif

29  const int tid = threadIdx.x + threadIdx.y * blockDim.x; // 0...63

32  __shared__ gather_t<CT, ORDER> shmem[32];
33 #if __CUDA_ARCH__ < 300
34 #ifdef DOMDEC_MSLDPME
35  __shared__ float4 shred_buf[32 * 2];
36  volatile float4 *shred = &shred_buf[(tid / 8) * 8];
37 #else
38  __shared__ float3 shred_buf[32 * 2];
39  volatile float3 *shred = &shred_buf[(tid / 8) * 8];
40 #endif
41 #endif

44  __shared__ double sh_flambda[32];
45 #endif

47  const int pos = blockIdx.x * blockDim.x + threadIdx.x;
48  const int pos_end = min((blockIdx.x + 1) * blockDim.x, ncoord);
49 
50  // Load atom data into shared memory
51  if (pos < pos_end && threadIdx.y == 0) {
52  float4 xyzqi = xyzq[pos];
53  float x = xyzqi.x;
54  float y = xyzqi.y;
55  float z = xyzqi.z;
56  float q = xyzqi.w;
57 
58  float w;
59 
60  w = x * recip1 + 2.0f;
61  float frx = (float)(nfftx * (w - (floorf(w + 0.5f) - 0.5f)));
62 
63  w = y * recip2 + 2.0f;
64  float fry = (float)(nffty * (w - (floorf(w + 0.5f) - 0.5f)));
65 
66  w = z * recip3 + 2.0f;
67  float frz = (float)(nfftz * (w - (floorf(w + 0.5f) - 0.5f)));
68 
69  int frxi = (int)frx;
70  int fryi = (int)fry;
71  int frzi = (int)frz;
72 
73  shmem[threadIdx.x].ix = frxi;
74  shmem[threadIdx.x].iy = fryi;
75  shmem[threadIdx.x].iz = frzi;
76  shmem[threadIdx.x].charge = q;
77 
78  float wx = frx - (float)frxi;
79  float wy = fry - (float)fryi;
80  float wz = frz - (float)frzi;
81 
82  float3 theta_tmp[ORDER];
83  float3 dtheta_tmp[ORDER];
84  calc_theta_dtheta<float, float3, ORDER>(wx, wy, wz, theta_tmp, dtheta_tmp);
85 
86 #pragma unroll
87  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].thetax[i] = theta_tmp[i].x;
88 
89 #pragma unroll
90  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].thetay[i] = theta_tmp[i].y;
91 
92 #pragma unroll
93  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].thetaz[i] = theta_tmp[i].z;
94 
95 #pragma unroll
96  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].dthetax[i] = dtheta_tmp[i].x;
97 
98 #pragma unroll
99  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].dthetay[i] = dtheta_tmp[i].y;
100 
101 #pragma unroll
102  for (int i = 0; i < ORDER; i++) shmem[threadIdx.x].dthetaz[i] = dtheta_tmp[i].z;
103  }
104  __syncthreads();
105 
106  // We divide the 6x6x6 cube into 8 3x3x3 sub-cubes.
107  // These sub-cubes are taken care by a single thread
108  //
109  // Calculate the index this thread is calculating
110  // tid = 0...63
111  const int t = (tid % 8); // 0...7
112  // t = (tx0 + ty0*2 + tz0*4)/3
113  // (tx0, ty0, tz0) gives the starting index of the 3x3x3 sub-cube
114  const int tz0 = (t / 4) * HALF_ORDER;
115  const int ty0 = ((t / 2) % 2) * HALF_ORDER;
116  const int tx0 = (t % 2) * HALF_ORDER;
117 
118  //
119  // Calculate forces for 32 atoms. We have 32*2 = 64 threads
120  // Loop is iterated 4 times:
121  // (iterations)
122  // Threads 0...7 = atoms 0, 8, 16, 24
123  // Threads 8...15 = atoms 1, 9, 17, 25
124  // Threads 16...31 = atoms 2, 10, 18, 26
125  // ...
126  // Threads 56...63 = atoms 7, 15, 23, 31
127  //
128 
129  int base = tid / 8;
130  const int base_end = pos_end - blockIdx.x * blockDim.x;
131  while (base < base_end) {
132 #ifdef DOMDEC_MSLDPME
133  double flambda = 0.0;
134 #endif
135 
136  float f1 = 0.0f;
137  float f2 = 0.0f;
138  float f3 = 0.0f;
139  int ix0 = shmem[base].ix;
140  int iy0 = shmem[base].iy;
141  int iz0 = shmem[base].iz;
142 
143 // Each thread calculates a 3x3x3 sub-cube
144 #pragma unroll
145  for (int i = 0; i < CUBE_HALF; i++) {
146  int tz = tz0 + (i / SQUARE_HALF);
147  int ty = ty0 + ((i / HALF_ORDER) % HALF_ORDER);
148  int tx = tx0 + (i % HALF_ORDER);
149 
150  int ix = ix0 + tx;
151  int iy = iy0 + ty;
152  int iz = iz0 + tz;
153 
154  if (ix >= nfftx) ix -= nfftx;
155  if (iy >= nffty) iy -= nffty;
156  if (iz >= nfftz) iz -= nfftz;
157 
158 #ifdef USE_TEXTURE_OBJECTS
159  float q0 = tex1Dfetch<float>(gridTexObj, ix + (iy + iz * ysize) * xsize);
160 #else
161  float q0 = tex1Dfetch(gridTexRef, ix + (iy + iz * ysize) * xsize);
162 #endif
163 
164  float thx0 = shmem[base].thetax[tx];
165  float thy0 = shmem[base].thetay[ty];
166  float thz0 = shmem[base].thetaz[tz];
167 
168  float dthx0 = shmem[base].dthetax[tx];
169  float dthy0 = shmem[base].dthetay[ty];
170  float dthz0 = shmem[base].dthetaz[tz];
171 
172  f1 += dthx0 * thy0 * thz0 * q0;
173  f2 += thx0 * dthy0 * thz0 * q0;
174  f3 += thx0 * thy0 * dthz0 * q0;
175 
176 #ifdef DOMDEC_MSLDPME
177  flambda += thx0 * thy0 * thz0 * q0;
178 #endif
179  }
180

183 // Reduce
184 #if __CUDA_ARCH__ >= 300
185  const int i = threadIdx.x & 7;
186 
187  f1 += SHFL(f1, i + 4, 8);
188  f2 += SHFL(f2, i + 4, 8);
189  f3 += SHFL(f3, i + 4, 8);
190 #ifdef DOMDEC_MSLDPME
191  flambda += SHFL(flambda, i + 4, 8);
192 #endif
193 
194  f1 += SHFL(f1, i + 2, 8);
195  f2 += SHFL(f2, i + 2, 8);
196  f3 += SHFL(f3, i + 2, 8);
197 #ifdef DOMDEC_MSLDPME
198  flambda += SHFL(flambda, i + 2, 8);
199 #endif
200 
201  f1 += SHFL(f1, i + 1, 8);
202  f2 += SHFL(f2, i + 1, 8);
203  f3 += SHFL(f3, i + 1, 8);
204 #ifdef DOMDEC_MSLDPME
205  flambda += SHFL(flambda, i + 1, 8);
206 #endif
207 
208  if (i == 0) {
209  shmem[base].f1 = f1;
210  shmem[base].f2 = f2;
211  shmem[base].f3 = f3;
212 #ifdef DOMDEC_MSLDPME
213  sh_flambda[base] = flambda;
214 #endif
215  }
216 
217 #else
218  const int i = threadIdx.x & 7;
219  shred[i].x = f1;
220  shred[i].y = f2;
221  shred[i].z = f3;
222 #ifdef DOMDEC_MSLDPME
223  shred[i].w = flambda;
224 #endif
225 
226  if (i < 4) {
227  shred[i].x += shred[i + 4].x;
228  shred[i].y += shred[i + 4].y;
229  shred[i].z += shred[i + 4].z;
230 #ifdef DOMDEC_MSLDPME
231  shred[i].w += shred[i + 4].w;
232 #endif
233  }
234 
235  if (i < 2) {
236  shred[i].x += shred[i + 2].x;
237  shred[i].y += shred[i + 2].y;
238  shred[i].z += shred[i + 2].z;
239 #ifdef DOMDEC_MSLDPME
240  shred[i].w += shred[i + 2].w;
241 #endif
242  }
243 
244  if (i == 0) {
245  shmem[base].f1 = shred[0].x + shred[1].x;
246  shmem[base].f2 = shred[0].y + shred[1].y;
247  shmem[base].f3 = shred[0].z + shred[1].z;
248 #ifdef DOMDEC_MSLDPME
249  sh_flambda[base] = shred[0].w + shred[1].w;
250 #endif
251  }
252 #endif
253 
254  base += 8;
255  }
256 
257  // Write forces
258  __syncthreads();
259  if (pos < pos_end && threadIdx.y == 0) {
260  float f1 = shmem[threadIdx.x].f1;
261  float f2 = shmem[threadIdx.x].f2;
262  float f3 = shmem[threadIdx.x].f3;
263 
264  float q = shmem[threadIdx.x].charge * ccelec;
265 
266 #ifdef DOMDEC_MSLDPME
267  // Ryan rearranged and scaled fx fy fz by bixlam and biflam by q
268  int iblock = blockIndexes[pos] & 0xffff;
269  if (iblock > 0) {
270  // Multiply by unscaled charge
271  atomicAdd(&biflam[iblock], q * sh_flambda[threadIdx.x]);
272  // Scale charge for spatial forces
273  q *= bixlam[iblock];
274  }
275 #endif
276 
277  float fx = q * recip1 * f1 * nfftx;
278  float fy = q * recip2 * f2 * nffty;
279  float fz = q * recip3 * f3 * nfftz;
280 
281  gather_force_store<FT>(fx, fy, fz, stride, pos, force);
282  }
283 }
